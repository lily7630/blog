函数防抖和函数节流 #

函数防抖 #

function debounce(fun, delay) {
  let timer;

  return function() {
    const context = this;
    const args = arguments;

    if (timer) {
      clearTimeout(timer);
    }
    timer = setTimeout(() => {
      fun && fun.apply(context, args);
    }, delay);
  }
}

// 调用方法
debounce(fn, 30)('参数');
